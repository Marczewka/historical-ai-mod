chinese_manchu_queue_hairstyle_character_trigger = { # hairsyle for Manchu characters not living living under discriminatory Han rule, and Han characters living under discriminatory Manchu rule
	exists = scope:character
	scope:character = {
		exists = home_country
		OR = {
			AND = {
				home_country = { 
					country_has_primary_culture = cu:manchu
					NOT = { any_primary_culture = { has_discrimination_trait = han_chinese } }
					cultural_acceptance_base = { target = cu:han value <= acceptance_status_4 }
				}
				culture = { 
					has_discrimination_trait = han_chinese 
				}
			}
			AND = {
				NOT = {
					home_country = { 
						any_primary_culture = { has_discrimination_trait = han_chinese }
						NOT = { country_has_primary_culture = cu:manchu }
						cultural_acceptance_base = { target = cu:manchu value >= acceptance_status_5 }
					}
				}
				culture = cu:manchu
			}
		}
	}
}
