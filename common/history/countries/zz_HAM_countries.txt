COUNTRIES = {
  # Lower max colonial affairs
  every_country = {
    limit = {
      OR = {
				has_law = law_type:law_colonial_exploitation
				has_law = law_type:law_colonial_resettlement
				has_law = law_type:law_frontier_colonization
			}
    }
    set_institution_investment_level = {
      institution = institution_colonial_affairs
      level = 1
    }
  }

  # Journals from day one
  every_country = {
    limit = {
      OR = {
        country_has_primary_culture = cu:north_german
        country_has_primary_culture = cu:south_german
      }
    }
    add_journal_entry = { type = je_german_unification_idea }
    add_journal_entry = { type = je_schleswig_holstein_question }
  }
  every_country = {
    limit = {
      can_form_nation = ETH
    }
    add_journal_entry = { type = je_age_of_princes }
  }
  every_country = {
    limit = {
      is_country_type = unrecognized
      has_technology_researched = colonization
      OR = {
        country_rank >= rank_value:unrecognized_major_power
        any_overlord_or_above = {
          country_rank >= rank_value:great_power
        }
      }
    }
    add_journal_entry = { type = je_earn_recognition }
  }

  c:CHL ?= {
    activate_law = law_type:law_no_colonial_affairs # colonizes too quickly
  }
  c:HAI ?= {
    set_tax_level = very_high
    set_government_wage_level = very_low
    set_military_wage_level = low
    add_taxed_goods = g:tobacco
  }
  c:RUS ?= {
    set_tax_level = medium # peasant taxes
  }
  c:JAP ?= {
    set_tax_level = low # better sol for peasants than in Russia
    set_military_wage_level = high # samurai stipends
  }
  c:USA ?= {
    add_journal_entry = { type = je_oregon }
    set_tariffs_export_priority = g:iron	
    set_tariffs_export_priority = g:steel	
    set_tariffs_export_priority = g:clothers
    set_tariffs_export_priority = g:luxury_clothes
    set_tariffs_export_priority = g:tools	
    set_tariffs_export_priority = g:glass
    set_tariffs_export_priority = g:furniture
    set_tariffs_export_priority = g:luxury_furniture
  }
  c:DEI ?= {
    activate_law = law_type:law_colonial_resettlement # already extraction economy
  }
}
