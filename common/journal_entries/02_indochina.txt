	}

	complete = {
		any_state = {
			filter = {

					state_region = s:STATE_LAOS
					state_region = s:STATE_CAMBODIA

				owner = { country_has_state_religion = rel:catholic }
				owner = { 
					overlord = { country_has_state_religion = rel:catholic } 
				}
			}

		}
	}

	on_complete = {
		trigger_event = { id = indochina.4 }
	}

	invalid = {
		NOT = { country_has_state_religion = rel:catholic }
	}
