# Root = Country
on_country_formed = {
	on_actions = {
		italian_homelands_on_action
	}
}

italian_homelands_on_action = {
	effect = {
		if = {
			limit = {
				c:ITA ?= this
				NOT = {
					cu:north_italian = {
						has_homeland = s:STATE_SAVOY
					}
				}
			}
			s:STATE_SAVOY = {
				add_homeland = cu:north_italian
			}
		}
	}
}

# Root = Diplomatic Play
on_diplo_play_start = {
	on_actions = {
		anti_subject_dp_liberty_desire_on_action
	}
}

anti_subject_dp_liberty_desire_on_action = {
	effect = {
		if = {
			limit = {
				OR = {
					is_diplomatic_play_type = dp_annex_subject
					is_diplomatic_play_type = dp_reduce_autonomy
				}
			}
			target ?= {
				add_liberty_desire = 100
			}
		}
	}
}
