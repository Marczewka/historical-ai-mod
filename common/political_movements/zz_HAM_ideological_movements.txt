movement_liberal = {
	category = movement_category_ideological
	
	ideology = ideology_liberal_movement
	character_ideologies = {
		ideology_market_liberal
		ideology_liberal_leader
		ideology_republican_leader
		ideology_radical
		ideology_humanitarian
	}
	
	creation_trigger = {
		has_technology_researched = democracy
	}	
	creation_weight = {
		value = 100
		if = {
			limit = {
				has_law = law_type:law_monarchy
				country_has_voting_franchise = no
			}
			multiply = {
				value = 0.1
			}
		}
		if = {
			limit = {
				owner ?= {
					has_journal_entry = je_springtime_of_the_peoples
				}
			}
			multiply = {
				value = 2
			}
		}
	}	
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}
	
	character_support_trigger = {
		trigger_if = {
			limit = {
				is_in_exile_pool = yes
			}
			OR = {
				has_ideology = ideology:ideology_market_liberal
				has_ideology = ideology:ideology_liberal_leader
				has_ideology = ideology:ideology_republican_leader
				has_ideology = ideology:ideology_radical
				has_ideology = ideology:ideology_humanitarian
				has_ideology = ideology:ideology_humanitarian_royalist	
			}
		}
		trigger_else = {
			OR = {
				has_ideology = ideology:ideology_market_liberal
				has_ideology = ideology:ideology_liberal_leader
				has_ideology = ideology:ideology_republican_leader
				has_ideology = ideology:ideology_radical
				has_ideology = ideology:ideology_humanitarian
				has_ideology = ideology:ideology_humanitarian_royalist
				interest_group ?= {
					OR = {
						has_ideology = ideology:ideology_liberal
						has_ideology = ideology:ideology_liberal_modern
					}
				}
			}
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_republican_leader
					has_ideology = ideology:ideology_liberal_leader
					has_ideology = ideology:ideology_market_liberal
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_radical
				NOT = {
					owner ?= {
						any_political_movement = {
							is_political_movement_type = movement_radical
						}
					}
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_humanitarian
				NOT = {
					is_interest_group_type = ig_trade_unions
				}
			}
			multiply = {
				value = 5
			}
		}
		if = { # Liberal agitators just inherently love liberalism
			limit = {
				interest_group ?= {
					OR = {
						has_ideology = ideology:ideology_liberal
						has_ideology = ideology:ideology_liberal_modern
					}
				}
			}
			multiply = {
				value = 2
			}
		}
	}

	pop_support_trigger = {
		NOT = {
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_urbanization
		movement_support_high_literacy
		movement_support_capitalists
		movement_support_aristocrats
		movement_support_shopkeepers
		movement_support_academics
		movement_support_bureaucrats
		movement_support_clerks
		movement_support_peasants
	}
	
	pop_support_weight = {	
		add = {
			value = 3
			desc = "POP_BASE_SUPPORT"
		}

		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}

		if = {
			limit = {
				strata = upper
			}
			if = {
				limit = {
					is_pop_type = capitalists
				}
				add = {
					value = 25
					desc = "POP_CAPITALISTS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = aristocrats
				}
				add = {
					value = 6
					desc = "POP_ARISTOCRATS"
				}
			}
			else = {
				add = {
					value = 9
					desc = "UPPER_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			if = {
				limit = {
					is_pop_type = shopkeepers
				}
				add = {
					value = 12
					desc = "POP_SHOPKEEPERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 24
					desc = "POP_ACADEMICS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = bureaucrats
				}
				add = {
					value = 12
					desc = "POP_BUREAUCRATS"
				}
			}
			else = {
				add = {
					value = 9
					desc = "MIDDLE_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 6
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = peasants
				}
				add = {
					value = 0
					desc = "POP_PEASANTS"
				}
			}
			else = {
				add = {
					value = 3
					desc = "LOWER_NO_ICON"
				}
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}
	}
	
	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		
		weight = {
			value = 100
		}	
	
		state_weight = {
			value = state_urbanization_rate	
		}
	}
	
	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 1000
		}

		state_weight = {
			value = state_urbanization_rate	
		}		
	}	
	
	additional_radicalism_factors = {
		owner = {
			if = {
				limit = { has_journal_entry = je_springtime_of_the_peoples }

				add = {
					value = 0.25
					desc = "SPRINGTIME_OF_THE_PEOPLES_UNDERWAY"
				}
			}

			if = {
				limit = { has_institution = institution_schools }
				
				add = {
					value = 3
					subtract = "institution_investment_level(institution_schools)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_schools"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_schools"
				}		
			}				
		}
	}		
}

movement_socialist = {
	category = movement_category_ideological
	
	ideology = ideology_socialist
	character_ideologies = {
		ideology_communist
		ideology_anarchist
		ideology_social_democrat
		ideology_vanguardist
	}
	
	creation_trigger = {
		has_technology_researched = socialism
	}	
	creation_weight = {
		value = 100
	}	
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}
	
	disband_trigger = {
		any_interest_group = {
			OR = {
				has_variable = chose_anarchism
				has_variable = chose_communism
				has_variable = chose_vanguardism
			}
		}
	}
	on_disbanded = {}
	
	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_communist
			has_ideology = ideology:ideology_anarchist
			has_ideology = ideology:ideology_social_democrat
			has_ideology = ideology:ideology_vanguardist
			AND = {
				is_interest_group_type = ig_trade_unions
				has_ideology = ideology:ideology_humanitarian
			}
		}
	}
	character_support_weight = { 
		value = 200
		if = { # Very vulnerable to fragmentation once the more focused movements emerge
			limit = {
				OR = {
					has_ideology = ideology:ideology_vanguardist
					has_ideology = ideology:ideology_communist
				}
				NOT = {
					owner ?= {
						any_political_movement = {
							is_political_movement_type = movement_communist
						}
					}
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_anarchist
				NOT = {
					owner ?= {
						any_political_movement = {
							is_political_movement_type = movement_anarchist
						}
					}
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_social_democrat
				NOT = {
					owner ?= {
						any_political_movement = {
							is_political_movement_type = movement_labor
						}
					}
				}
			}
			multiply = {
				value = 5
			}
		}
	}
	
	pop_support_trigger = {
		NOR = {
			is_pop_type = capitalists
			is_pop_type = aristocrats
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_urbanization
		movement_support_high_literacy
		movement_support_below_expected_sol
		movement_support_radicalization
		movement_support_academics
		movement_support_laborers
		movement_support_machinists
		movement_support_soldiers_conscripts
	}

	pop_support_weight = {	
		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}	
	
		if = {
			limit = {
				strata = middle
			}
			multiply = {
				value = 0.10
				desc = "MIDDLE_NO_ICON"
			}
			if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 3
					desc = "POP_ACADEMICS"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = laborers
				}
				add = {
					value = 18
					desc = "POP_LABORERS"
				}
			}		
			else_if = {
				limit = {
					is_pop_type = machinists
				}
				add = {
					value = 24
					desc = "POP_MACHINISTS"
				}
			}
			else = {
				add = {
					value = 6
					desc = "LOWER_NO_ICON"
				}
			}

			if = {
				limit = {
					AND = {
						is_pop_type = soldiers
						pop_employment_building = building_conscription_center
					}
				}
				multiply = {
					value = 1.5
					desc = "CONSCRIPTS"
				}
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
				strata = lower
			}
			add = { # Increases likelihood of support if impoverished
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}

			if = {
				limit = {
					owner = {
						NOR = { 
							has_law = law_type:law_council_republic 
							has_law = law_type:law_universal_suffrage
						}
					}
				}
				multiply = 2.0
			}		
		}
		else_if = {
			limit = {
				standard_of_living > state.average_expected_sol
				strata = lower
			}
			add = { # Decreases likelihood of support if well-off
				desc = "IS_ABOVE_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = { # Multiplied by 1.5x max
			desc = "POP_RADICALS"
			value = pop_radical_fraction
			divide = 2
			add = 1
		}

		if = {
			limit = {
				owner = {
					any_political_movement = {
						count >= 2

						OR = {
							is_political_movement_type = movement_labor
							is_political_movement_type = movement_communist
							is_political_movement_type = movement_anarchist
						}
						political_movement_support >= 0.10
					}
				}
			}
			multiply = 0.05
		}	
	}
	
	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		
		weight = {
			value = 100
		}	
	
		state_weight = {
			value = state_urbanization_rate
			
			if = {
				limit = { is_capital = yes }
				if = {
					limit = { owner ?= { has_law = law_type:law_council_republic } }
					multiply = 10.0
				}
				else = {
					multiply = 0.1
				}
			}			
		}
	}
	
	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 1000
		}

		state_weight = {
			value = state_urbanization_rate
			
			if = {
				limit = { is_capital = yes }
				if = {
					limit = { owner ?= { has_law = law_type:law_council_republic } }
					multiply = 10.0
				}
				else = {
					multiply = 0.1
				}
			}			
		}		
	}	
	
	additional_radicalism_factors = {
		owner = {
			if = {
				limit = { has_institution = institution_social_security }
				
				add = {
					value = 3
					subtract = "institution_investment_level(institution_social_security)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_social_security"
				}
			}
			else = {
				add = {
					value = 0.1

			multiply = 10
		}

		if = {
			limit = {
				strata = middle
			}
			multiply = {
				value = 0.10
				desc = "MIDDLE_NO_ICON"
			}
			if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 3
					desc = "POP_ACADEMICS"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower

				limit = {
					is_pop_type = laborers
				}
				add = {
					value = 15
					desc = "POP_LABORERS"

					value = 15
					desc = "POP_PEASANTS"
				}
			}
			else = {
				add = {
					value = 6
					desc = "LOWER_NO_ICON"
				}
			}
			if = {
				limit = {
					OR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
						pop_employment_building_group = bg_logging
						pop_employment_building_group = bg_fishing
						pop_employment_building_group = bg_whaling
						pop_employment_building_group = bg_manor_houses
					}
				}
				multiply = {
					value = 1.25
					desc = "RURAL_POP"
				}
			}

			if = {
				limit = {
					AND = {
						is_pop_type = soldiers
						pop_employment_building = building_conscription_center
					}
				}
				multiply = {
					value = 1.15
					desc = "CONSCRIPTS"
				}
			}
		}	

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
				strata = lower # Impoverished middle class pops will support fascists instead
			}
			add = { # Increases likelihood of support if impoverished
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 15
			}

			if = {
				limit = {
					owner = {
						NOR = {
							has_law = law_type:law_homesteading
							has_law = law_type:law_collectivized_agriculture
						}
					}
				}
				multiply = 1.5
			}
			if = {
				limit = {
					owner = {
						NOR = {
							has_law = law_type:law_anarchy
							has_law = law_type:law_universal_suffrage
						}				
					}
				}
				multiply = 1.5
			}				
		}
		else_if = {
			limit = {
				standard_of_living > state.average_expected_sol
				strata = lower
			}
			add = { # Decreases likelihood of support if well-off
				desc = "IS_ABOVE_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
			}
		}

			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = { # Multiplied by 2x max
			desc = "POP_RADICALS"
			value = pop_radical_fraction
			add = 1
		}	
	}

		}

			value = 100
		}	
	}

			OR = {
				scope:clout <= 0
				AND = {
					scope:clout > 0
					scope:clout <= 0.15
					owner ?= {
						NOR = { 
							has_law = law_type:law_council_republic
							is_enacting_law = law_type:law_council_republic
							is_enacting_law = law_type:law_anarchy
							any_interest_group = {
								is_in_government = yes
								leader ?= {
									has_ideology = ideology:ideology_anarchist
								}
							}
						} 
					}
				}
			}
		}
		
		weight = {
			value = 1000
		}		
	}

				value = 15
				subtract = average_sol
				multiply = 0.03 # 0.025
				min = 0
			}
		}
	}
