je_sell_alaska_button = {
	name = "je_sell_alaska_button"
	desc = "je_sell_alaska_button_desc"

	visible = { 
		c:ALK ?= {
			is_player = no
		} 
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = { 
				scaled_debt > 0
			}
			add = 25
		}
		modifier = {
			trigger = { 
				has_modifier = declared_bankruptcy
			}
			add = 50
		}
		modifier = {
			trigger = { 
				any_country = {
					NOT = { this = root }
					country_rank >= root.country_rank
					has_interest_marker_in_region = sr:region_pacific_coast
					relations:root <= relations_threshold:poor
				}
			}
			add = 25
		}
		modifier = {
			trigger = { 
				any_country = {
					NOT = { this = root }
					country_rank >= root.country_rank
					has_interest_marker_in_region = sr:region_pacific_coast
					relations:root <= relations_threshold:cold
				}
			}
			add = 50
		}
	}

	possible = {
		custom_tooltip = {
			text = has_alaska_sale_cooldown_tt
			NOT = { has_variable = alaska_sale_cooldown }
		}
		OR = {
			has_state_in_state_region = STATE_ALASKA
			any_subject_or_below = {
				has_state_in_state_region = STATE_ALASKA
			}
		}
		custom_tooltip = {
			text = friendly_nation_with_interest_in_alaska_tt
			any_country = {
				NOT = { is_subject = yes }
				NOT = { this = root }
				country_is_in_north_america = yes
				relations:root > relations_threshold:poor
				country_rank >= rank_value:great_power
				any_scope_state = {
					region = sr:region_pacific_coast
					is_incorporated = yes
				}
			}
		}
	}

	effect = {
		set_variable = {
			name = alaska_sale_cooldown
			months = 12
		}
		save_scope_as = alaska_seller
		random_country = {
			limit = {
				NOT = { is_subject = yes }
				NOT = { this = root }
				country_is_in_north_america = yes
				relations:root > relations_threshold:poor
				country_rank >= rank_value:great_power
				any_scope_state = {
					region = sr:region_pacific_coast
					is_incorporated = yes
				}
			}
			trigger_event = { id = alaska.4 popup = yes }
		}
	}
}

		}
		country_rank >= rank_value:minor_power
		sr:region_italy = {
			any_scope_country = {
				NOR = {
					country_has_primary_culture = cu:south_italian
					country_has_primary_culture = cu:north_italian
				}
				custom_tooltip = {
					text = owns_italian_homelands_tt
					any_scope_state = {
						state_region = {
							OR = {
								is_homeland = cu:south_italian
								is_homeland = cu:north_italian
							}
						}
					}
				}
				relations:root <= 20
				NOT = {
					any_political_movement = {
						has_modifier = austria_italian_radicals
					}
				}
			}
		}
	}

	ai_chance = {
		base = 50
		modifier = {
			trigger = {
				has_law = law_type:law_theocracy
			}
			add = -70
		}
		modifier = {
			trigger = {
				any_country = {
					NOR = {
						country_has_primary_culture = cu:south_italian
						country_has_primary_culture = cu:north_italian
					}
					any_scope_state = {
						state_region = {
							OR = {
								is_homeland = cu:south_italian
								is_homeland = cu:north_italian
							}
						}
					}
					relations:root <= 20
					NOT = {
						any_political_movement = {
							is_political_movement_type = movement_cultural_minority
							OR = {
								culture = cu:north_italian
								culture = cu:south_italian
							}
							has_modifier = austria_italian_radicals
						}
					}
				}
			}
			add = 25
		}
	}

	effect = {
		custom_tooltip = je_risorgimento_austrian_radicals_button
		ROOT = {
			save_scope_as = saboteur_country
		}
		every_country = {
			limit = {
				NOR = {
					country_has_primary_culture = cu:south_italian
					country_has_primary_culture = cu:north_italian
				}
				any_scope_state = {
					state_region = {
						OR = {
							is_homeland = cu:south_italian
							is_homeland = cu:north_italian
						}
					}
				}
				relations:root <= 20
				NOT = {
					any_political_movement = {
						is_political_movement_type = movement_cultural_minority
						OR = {
							culture = cu:north_italian
							culture = cu:south_italian
						}
						has_modifier = austria_italian_radicals
					}
				}
			}
			every_scope_state = {
				limit = {
					state_region = { 
						is_homeland = cu:north_italian
					}
				}
				add_radicals_in_state = {
					value = 0.10
					culture = cu:north_italian
				}
			}
			every_scope_state = {
				limit = {
					state_region = {
						is_homeland = cu:south_italian
					}
				}
				add_radicals_in_state = {
					value = 0.10
					culture = cu:south_italian
				}
			}
			if = {
				limit = {
					NOT = {
						any_political_movement = {
							is_political_movement_type = movement_cultural_minority
							OR = {
								culture = cu:north_italian
								culture = cu:south_italian
							}
						}
					}
				}
				create_political_movement = { 
					type = movement_cultural_minority 
					culture = cu:north_italian
				}
			}
			random_political_movement = {
				limit = {
					is_political_movement_type = movement_cultural_minority
					OR = {
						culture = cu:north_italian
						culture = cu:south_italian
					}
					NOT = { has_modifier = austria_italian_radicals }
				}
				add_modifier = {
					name = austria_italian_radicals
					months = short_modifier_time
				}
			}
			THIS = {
				save_scope_as = victim_country
			}
			post_notification = nationalism_decision_agitation_occurred
			every_country = {
				limit = {
					has_interest_marker_in_region = region_italy
					NOT = { this = prev }
				}
				post_notification = nationalism_decision_agitation_stirred
			}
		}
	}
}

			}

	}

		base = 100
	}

	}
