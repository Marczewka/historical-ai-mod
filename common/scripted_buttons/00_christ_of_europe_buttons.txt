christ_of_nations_button = {
	name = "christ_of_nations_button"
	desc = "christ_of_nations_button_desc"

	visible = { always = yes }

	possible = {
        is_subject = no
		country_rank >= rank_value:minor_power
		sr:region_poland = {
			any_scope_country = {
				NOT = { country_has_primary_culture = cu:polish }
				custom_tooltip = {
					text = owns_polish_homelands_tt
					any_scope_state = {
						state_region = {
							OR = {
								is_homeland = cu:polish
							}
						}
					}
				}
				relations:root <= 20
				NOT = {
					any_political_movement = {
						has_modifier = polish_radicals
					}
				}
			}
		}
	}

	effect = {
		custom_tooltip = christ_of_nations_button
		ROOT = {
			save_scope_as = saboteur_country
		}
		every_country = {
			limit = {
				NOT = { country_has_primary_culture = cu:polish }
				any_scope_state = {
					state_region = {
						OR = {
							is_homeland = cu:polish
						}
					}
				}
				relations:root <= 20
				NOT = {
					any_political_movement = {
						is_political_movement_type = movement_cultural_minority
						OR = {
							culture = cu:polish
						}
						has_modifier = polish_radicals
					}
				}
			}
			every_scope_state = {
				limit = {
					state_region = { 
						is_homeland = cu:polish
					}
				}
				add_radicals_in_state = {
					value = 0.10
					culture = cu:polish
				}
				set_variable = {
					name = christ_of_nations_target
					months = long_modifier_time
				}
			}
			if = {
				limit = {
					NOT = {
						any_political_movement = {
							is_political_movement_type = movement_cultural_minority
							OR = {
								culture = cu:polish
							}
						}
					}
				}
				create_political_movement = { 
					type = movement_cultural_minority 
					culture = cu:polish
				}
			}
			random_political_movement = {
				limit = {
					is_political_movement_type = movement_cultural_minority
					OR = {
						culture = cu:polish
					}
					NOT = { has_modifier = polish_radicals }
				}
				add_modifier = {
					name = polish_radicals
					months = short_modifier_time
				}
			}
			THIS = {
				save_scope_as = victim_country
			}
			post_notification = polish_decision_agitation_stirred
			every_country = {
				limit = {
					has_interest_marker_in_region = region_poland
					NOT = { this = prev }
				}
				post_notification = polish_decision_agitation_stirred
			}
		}
	}
}

je_christ_of_nations_unification_button = {
	name = "je_christ_of_nations_unification_button"
	desc = "je_christ_of_nations_unification_button_desc"

	visible = { 
		NOT = {
			is_subject = yes
		}
	}

	possible = {
		has_technology_researched = nationalism
		NOR = {
			is_revolutionary = yes
			is_secessionist = yes
		}
		custom_tooltip = {
			text = rejected_polish_union_pol_var_tt
			NOT = { has_variable = rejected_polish_union_pol_var }
		}
		any_country = {
			NOT = { this = root }
			NOR = {
				is_revolutionary = yes
				is_secessionist = yes
			}
			country_has_primary_culture = cu:polish
			OR = {
				is_subject = no
				is_subject_of = ROOT
			}
			OR = {
				this.gdp < root.gdp
				root = { is_player = yes }
			}
			relations:root >= relations_threshold:amicable
			is_player = no
		}
	}

	ai_chance = {
		base = 100
	}

	effect = {
		trigger_event = { id = poland_events.3 popup = yes }
	}
}
