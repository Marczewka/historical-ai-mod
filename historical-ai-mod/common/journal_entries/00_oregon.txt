je_oregon = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_usa_manifest_destiny

	is_shown_when_inactive = {
		exists = c:USA
		exists = c:GBR
		c:USA = ROOT
		s:STATE_WASHINGTON = {
			any_scope_state = {
				owner = {
					OR = {
						c:GBR = THIS
						is_subject_of = c:GBR
					}
				}
			}
		}
		s:STATE_OREGON = {
			any_scope_state = {
				owner = {
					OR = {
						c:GBR = THIS
						is_subject_of = c:GBR
					}
				}
			}
		}
		s:STATE_IDAHO = {
			any_scope_state = {
				owner = {
					OR = {
						c:GBR = THIS
						is_subject_of = c:GBR
					}
				}
			}
		}
	}

	possible = {
		# Colonies don't generate interest
		always = yes
		# OR = {
		# 	has_interest_marker_in_region = sr:region_pacific_coast
		# 	custom_tooltip = {
		# 		text = oregon_tt
		# 		has_global_variable = oregon_trail_mapped
		# 	}
		# 	owns_entire_state_region = STATE_MONTANA
		# 	owns_entire_state_region = STATE_WYOMING
		# }
	}

	complete = {
		c:GBR ?= {
			relations:ROOT > relations_threshold:poor
		}
        has_technology_researched = nationalism
		any_scope_state = {
			OR = {
				state_region = s:STATE_MONTANA
				state_region = s:STATE_WYOMING
			}
		}
	}

	on_complete = {
		c:GBR ?= { trigger_event = { id = oregon.2 } }
	}

	# Lowered requirements for completion but added fail state
	fail = {
		c:GBR = {
			relations:ROOT <= relations_threshold:cold
		}
	}

	on_fail = {
		c:USA = {
			trigger_event = {
				id = oregon.4
				days = 0
			}
		}
	}

	# Changed direct acquisition from complete to invalid state
	invalid = {
		OR = {
			AND = {
				owns_entire_state_region = STATE_WASHINGTON
				owns_entire_state_region = STATE_IDAHO
				owns_entire_state_region = STATE_OREGON
			}
			c:ORG ?= {
				is_subject_of = ROOT
			}
		}
	}

	on_invalid = {
		s:STATE_IDAHO = {
			add_homeland = cu:yankee
		}
		s:STATE_WASHINGTON = {
			add_homeland = cu:yankee
		}
		s:STATE_OREGON = {
			add_homeland = cu:yankee
		}
		if = {
			limit = {
				s:STATE_BRITISH_COLUMBIA = {
					any_scope_state = {
						owner = {
							OR = {
								c:USA ?= THIS
								is_subject_of = c:USA
							}
						}
					}
				}
			}
			s:STATE_BRITISH_COLUMBIA = {
				add_homeland = cu:yankee
			}
		}
		add_modifier = {
			name = oregon_resolved
			days = normal_modifier_time
		}
	}

	weight = 100
}
