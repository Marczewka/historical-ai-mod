REPLACE:can_reach_target_country = {
	has_diplomatic_relevance = scope:target_country
	OR = {
		has_strategic_adjacency = scope:target_country
		AND = {
			has_port = yes
			navy_size >= 10
			scope:target_country = { 
				any_scope_state = {
					is_coastal = yes 
				}
			}
		}
	}
}

REPLACE:can_be_reached_by_target_country = {
	has_diplomatic_relevance = scope:target_country
	OR = {
		has_strategic_adjacency = scope:target_country
		AND = {
			any_scope_state = {
				is_coastal = yes 
			}
			scope:target_country = { has_port = yes }
			scope:target_country = { navy_size >= 10 }
		}
	}
}

REPLACE:can_reach_target_state = {
	has_diplomatic_relevance = scope:target_state.owner
	OR = {
		has_strategic_adjacency = scope:target_state
		AND = {
			navy_size >= 10
			has_port = yes
			scope:target_state = { is_coastal = yes }
		}
	}
}

REPLACE:is_oppressor_of_schleswig_holstein = {
	OR = {
		top_overlord ?= {
			NOR = {
				country_has_primary_culture = cu:north_german
				country_has_primary_culture = cu:south_german
			}
		}
		NOR = {
			country_has_primary_culture = cu:north_german
			country_has_primary_culture = cu:south_german
		}
	}
	OR = {
		any_scope_state = {
			owner = prev
			state_region = s:STATE_SCHLESWIG_HOLSTEIN
		}
		any_subject_or_below = {
			any_scope_state = {
				owner = prev
				state_region = s:STATE_SCHLESWIG_HOLSTEIN
			}
		}
	}
}
