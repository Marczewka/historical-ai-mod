REPLACE:movement_italian_national = {
	category = movement_category_pan_national

	ideology = ideology_national_liberal_movement
	character_ideologies = {
		ideology_ethno_nationalist
		ideology_market_liberal
		ideology_liberal_leader
		ideology_republican_leader
		ideology_radical
	}

	can_cause_obstinance = yes

	creation_trigger = {
		is_in_geographic_region = geographic_region_italy_old
		OR = {
			country_has_primary_culture = cu:north_italian
			country_has_primary_culture = cu:south_italian
		}
		NOR = {
			c:AUS ?= this
			c:KUK ?= this
			c:HRE ?= this
			c:ITA ?= this
		}
		OR = {
			cu:north_italian.culture_current_fervor > 10
			cu:south_italian.culture_current_fervor > 10
		}
	}
	creation_weight = {
		value = 100
		if = {
			limit = {
				owner ?= { # If a movement of this type exists in a neighbouring country, make it more likely to spawn here as well
					any_neighbouring_state = {
						owner = {
							prestige > ROOT.owner.prestige
							any_political_movement = {
								is_political_movement_type = movement_italian_national
							}
						}
					}
				}
			}
			multiply = {
				value = 2
			}
		}
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}

	disband_trigger = {
		c:ITA ?= this
	}

	on_disbanded = {
		if = {
			limit = {
				c:ITA ?= owner
			}
			owner ?= {
				create_political_movement = { type = movement_cultural_majority }
			}
		}
	}

	character_support_trigger = {
		culture = {
			has_discrimination_trait = heritage_italic
		}
	}
	character_support_weight = {
		value = 100
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_ethno_nationalist
					has_liberal_ideology = yes
					has_progressive_ideology = yes
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_conservative_ideology = yes
			}
			multiply = {
				value = 0.5
			}
		}
		else_if = {
			limit = {
				has_reactionary_ideology = yes
				NOT = {
					has_ideology = ideology:ideology_ethno_nationalist
				}
			}
			multiply = {
				value = 0
			}
		}
	}

	pop_support_trigger = {
		culture = {
			has_discrimination_trait = heritage_italic
		}
	}

	pop_support_factors = {
		movement_support_national_awakening
		movement_support_cultural_fervor
		movement_support_high_urbanization
		movement_support_italic_heritage
		movement_support_high_literacy
		movement_support_academics
		movement_support_clerks
		movement_support_shopkeepers
		movement_support_officers
		movement_support_soldiers
		movement_support_nationalism_tech
		movement_support_pan_nationalism_tech
	}

	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
					OR = {
						is_being_suppressed = yes
						is_being_bolstered = yes
					}
				}
			}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}

		if = {
			limit = {
				strata = upper
			}
			if = {
				limit = {
					is_pop_type = capitalists
				}
				add = {
					value = 10
					desc = "POP_CAPITALISTS"
				}
			}
			else = {
				add = {
					value = 4
					desc = "UPPER_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 35
					desc = "POP_ACADEMICS"
				}
			}
			if = {
				limit = {
					is_pop_type = shopkeepers
				}
				add = {
					value = 25
					desc = "POP_SHOPKEEPERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = officers
				}
				add = {
					value = 15
					desc = "POP_OFFICERS"
				}
			}
			else = {
				add = {
					value = 12
					desc = "MIDDLE_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 20
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 10
					desc = "POP_SOLDIERS"
				}
			}
			else = {
				add = {
					value = 5
					desc = "LOWER_NO_ICON"
				}
			}
		}

		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}

		if = {
			limit = {
				owner ?= {
					has_technology_researched = pan-nationalism
				}
			}
			multiply = {
				value = 1.5
				desc = "PAN_NATIONALISM_TECH"
			}
		}
		else_if = {
			limit = {
				owner ?= {
					has_technology_researched = nationalism
				}
			}
			multiply = {
				value = 1.25
				desc = "NATIONALISM_TECH"
			}
		}

		multiply = { # Range from x0.775 to 2.275
			desc = "POP_CULTURE_FERVOR"
			value = culture.culture_current_fervor
			subtract = 15
			multiply = 0.015
			add = 1
		}

		if = {
			limit = {
				culture ?= {
					culture_has_national_awakening = yes
				}
			}
			multiply = {
				value = 1.5
				desc = "POP_ACTIVE_NATIONAL_AWAKENING"
			}
		}

		if = { # If a more radical movement exists, it'll poach radicals from this one.
			limit = {
				owner ?= {
					any_political_movement = {
						OR = {
							is_political_movement_type = movement_radical
							is_political_movement_type = movement_fascist
						}
					}
				}
			}
			multiply = {
				desc = "POP_RADICAL_MOVEMENT_LOSSES"
				value = 1
				subtract = pop_radical_fraction
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}
	}
	revolution = {
		possible = {
			political_movement_support >= 0.05
		}

		weight = {
			value = 100
		}

		state_weight = {
			value = 0.25
			add = scope:clout
			add = { value = this.turmoil divide = 2 }
			subtract = this.loyalists
			if = {
			   limit = {
					NOT = {
						state_is_homeland_of_italian_cultures = yes
					}
				}
				multiply = 0
			}
		}
	}

	secession = {
		possible = {
			owner ?= {
				has_law_or_variant = law_type:law_subjecthood
			}
			political_movement_support >= 0.05
			scope:clout <= 0
		}

		weight = {
			value = 1000
		}

		state_weight = {
			value = 0.25
			add = { value = this.turmoil divide = 2 }
			subtract = this.loyalists
			if = {
			   limit = {
					NOT = {
						state_is_homeland_of_italian_cultures = yes
					}
				}
				multiply = 0
			}
		}

		forced_tags = {
			ITA = {
				trigger = {
					always = yes
				}
				weight = 1000
			}
		}
	}

	additional_radicalism_factors = {
		if = {
			limit = {
				owner.ruler ?= {
					NOT = { culture = { is_primary_culture_of = root.owner } }
				}
			}

			add = {
				value = 0.1
				desc = "RULER_NOT_PRIMARY_CULTURE"
			}
		}
		if = {
			limit = {
				owner ?= {
					has_technology_researched = nationalism
				}
			}
			if = {
				limit = {
					owner ?= {
						is_unification_candidate = ITA
						any_diplomatically_relevant_country = {
							is_unification_candidate = ITA
						}
					}
				}
				add = {
					desc = "RIVAL_ITALIAN_UNIFIERS"
					value = 0.10
				}
			}
			else_if = {
				limit = {
					owner ?= {
						NOT = {
							is_unification_candidate = ITA
						}
						any_diplomatically_relevant_country = {
							is_unification_candidate = ITA
						}
					}
				}
				add = {
					desc = "SUPPORT_FOR_UNIFIER"
					value = 0.10
				}
			}
		}
		if = {
			limit = {
				owner ?= {
					has_technology_researched = pan-nationalism
				}
			}
			if = {
				limit = {
					owner ?= {
						is_unification_candidate = ITA
						NOT = {
							any_state_region = {
								OR = {
									is_homeland = cu:south_italian
									is_homeland = cu:north_italian
								}
								any_scope_state = {
									NOT = {
										owner = ROOT.owner
									}
								}
							}
						}
					}
				}
				add = {
					desc = "ITALIAN_UNIFICATION"
					value = 0.10
				}
			}
		}

		if = {
			limit = {
				owner = {
					any_neighbouring_state = {
						owner = {
							NOT = { THIS = ROOT.owner }
							any_political_movement = {
								is_political_movement_type = movement_italian_national
								is_insurrectionary = yes
							}
						}
					}
				}
			}
			add = {
				desc = "NEIGHBOURING_COUNTRY_WITH_REVOLUTION"
				value = 0.15
			}
		}

		add = {
			desc = "FERVOR_OF_ITALIAN_CULTURE_ACTIVISM"
			value = owner.average_italian_fervor
			divide = 400
		}
		if = {
			limit = {
				this.days_since_movement_defeated <= very_long_modifier_time
			}
			add = { # (x-y)/y - ([1, 7300]-7300)/7300
				desc = "movement_suppressed_after_civil_war"
				value = this.days_since_movement_defeated
				subtract = very_long_modifier_time
				divide = {
					value = very_long_modifier_time
				}
			}
		}
		if = {
			limit = {
				owner = {
					has_journal_entry = je_risorgimento
				}
			}
			add = {
				desc = "RISORGIMENTO_UNDERWAY"
				value = 0.10
			}
		}

		if = {
			limit = {
				owner = {
					has_variable = risorgimento_anti_unifying_stance
				}
			}
			add = {
				desc = "OPPOSING_RISORGIMENTO"
				value = 0.10
			}
		}
	}
}
